	value = 0
	ROOT = {
		every_country_culture = {
			limit = {
				NOT = { this = scope:target_culture }
				this.culture.culture_group = scope:target_culture.culture.culture_group
			}
			add = country_culture_pop_count
		}
	}
	divide = 4
	round = floor
	min = 80
	max = 250
}
